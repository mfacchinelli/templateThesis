% Command to cite frame as X-frame and add mouse-hover tag
\newcommand{\framename}{}
\newcommand{\citeframe}[1]{%
	% Get name
	\ifthenelse{\equal{#1}{I}}{%
		\renewcommand{\framename}{Inertial Planetocentric}}{%
	\ifthenelse{\equal{#1}{O}}{%
		\renewcommand{\framename}{Rotating Planetocentric}}{%
	\ifthenelse{\equal{#1}{V}}{%
		\renewcommand{\framename}{Vertical}}{%
	\ifthenelse{\equal{#1}{T}}{%
		\renewcommand{\framename}{Body-fixed}}{%
	}}}}%
	% Print name
	\pdftooltip{\framefont{#1}-frame}{\framename\ Reference Frame}%
}

% Command to cite frame as F_X and add mouse-hover tag
\newcommand{\citefframe}[1]{%
	% Get name
	\ifthenelse{\equal{#1}{I}}{%
		\renewcommand{\framename}{Inertial Planetocentric}}{%
	\ifthenelse{\equal{#1}{O}}{%
		\renewcommand{\framename}{Rotating Planetocentric}}{%
	\ifthenelse{\equal{#1}{V}}{%
		\renewcommand{\framename}{Vertical}}{%
	\ifthenelse{\equal{#1}{T}}{%
		\renewcommand{\framename}{Body-fixed}}{%
	}}}}%
	% Print name
	\pdftooltip{\framefont{F_{#1}}}{\framename\ Reference Frame}%
}
